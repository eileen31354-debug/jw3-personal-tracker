如何配置 Apps Script？
1. 打开 Google Sheet → 扩展程序 → Apps 脚本
2. 粘贴以下代码并保存：

var SECRET='YOUR_SECRET_KEY';function doGet(e){if(e.parameter.key!==SECRET)return ContentService.createTextOutput('{"err":"unauthorized"}').setMimeType(ContentService.MimeType.JSON);var s=SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();var v=s.getRange('A1').getValue();return ContentService.createTextOutput(v||'{}').setMimeType(ContentService.MimeType.JSON);}function doPost(e){try{var d=JSON.parse(e.postData.contents);if(d._key!==SECRET)return ContentService.createTextOutput('{"err":"unauthorized"}').setMimeType(ContentService.MimeType.JSON);delete d._key;var s=SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();s.getRange('A1').setValue(JSON.stringify(d));return ContentService.createTextOutput('{"ok":true}').setMimeType(ContentService.MimeType.JSON);}catch(err){return ContentService.createTextOutput('{"err":"'+err.message+'"}').setMimeType(ContentService.MimeType.JSON);}}

3. 部署 → 新建部署 → 类型：网络应用 → 权限：任何人
4. 复制 Web App URL 填入上方输入框
